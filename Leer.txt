CREATE TABLE Productos (
    IdProducto INT IDENTITY(1,1) PRIMARY KEY,  
    Nombre NVARCHAR(100) NOT NULL,            
    Descripcion NVARCHAR(255),           
    Precio DECIMAL(18, 2) NOT NULL,          
    Cantidad INT NOT NULL,                    
    Categoria NVARCHAR(50) NOT NULL,

);
